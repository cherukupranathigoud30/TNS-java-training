package conser;

public class TypeCastingDemo {

	public static void main(String[] args) {
		//implict typecasting
		byte b = 45;
		int i1 =b;
		System.out.println("i1=" +i1);
		float f= 1.2f;
		double d1= f;
		System.out.println("d1=" +d1);
		char ch='A';
		int i2 =ch;
		System.out.println("i2=" +i2);
		//explict typecasting
		double dob=4.567f;
		long l1=(long)dob;
		System.out.println("l1=" +l1);
		float f2 = 5.4f;
		int i=(int) f2;
		System.out.println(i);
	}

}
