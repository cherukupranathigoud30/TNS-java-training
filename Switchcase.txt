package conser;

public class SwitchCase {

	public static void main(String[] args) {
		
				char x = 'l';
				switch (x) 
				{
				case 'l' : 
				case 'L' : 
					System.out.println(x+" is a Letter");
					break;
				case 'd':
				case 'D' :
					System.out.println(x+" is a Digit");
					break;	
				
				default:
					System.out.println(x+" is other than letter, digit, space or special symbol ");
					break;
				}
			
			}
